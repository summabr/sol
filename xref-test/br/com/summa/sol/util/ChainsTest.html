45      @Test
46      public void testTraverse() {
47          List<String> notes = Arrays.asList("do", "re", "mi");
48          assertEquals("do/re/mi",
49                  Chains.traverse("/", notes));
50          assertEquals("do/re/mi",
51                  Chains.traverse("/", notes.iterator()));
52          List<Integer> primes = Arrays.asList(2, 3, 5, 7, 11, 13, 17);
53          assertEquals("2 < 3 < 5 < 7 < 11 < 13 < 17",
54                  Chains.traverse(" < ", primes));
55          assertEquals("2 < 3 < 5 < 7 < 11 < 13 < 17",
56                  Chains.traverse(" < ", primes.iterator()));
57          List<?> list = Arrays.asList(false, 1, "two", 3L, "IV", null, '6');
58          assertEquals("false, 1, two, 3, IV, null, 6",
59                  Chains.traverse(", ", list));
60          assertEquals("false, 1, two, 3, IV, null, 6",
61                  Chains.traverse(", ", list.iterator()));
62          List<String> list2 = Arrays.asList("", "", null, "a", "");
63          assertEquals("**null*a*",
64                  Chains.traverse("*", list2));
65          assertEquals("..null.a.",
66                  Chains.traverse(".", list2.iterator()));
67          LinkedList<Character> list3 = new LinkedList<Character>(Arrays.asList('a', 'b', 'c'));
68          assertEquals("c - b - a",
69                  Chains.traverse(" - ", list3.descendingIterator()));
70      }

72      @Test
73      public void testMap() {
74          Map<String, Object> codes = new LinkedHashMap<String, Object>();
75          codes.put("USA", 1);
76          codes.put("UK", 44);
77          codes.put("Brazil", 55);
78  
79          assertEquals("USA=1, UK=44, Brazil=55",
80                  Chains.traverse(", ", codes.entrySet()));
81          assertEquals("USA=1, UK=44, Brazil=55",
82                  Chains.traverse(", ", codes.entrySet().iterator()));
83          assertEquals("USA, UK, Brazil",
84                  Chains.traverse(", ", codes.keySet()));
85          assertEquals("USA, UK, Brazil",
86                  Chains.traverse(", ", codes.keySet().iterator()));
87          assertEquals("1, 44, 55",
88                  Chains.traverse(", ", codes.values()));
89          assertEquals("1, 44, 55",
90                  Chains.traverse(", ", codes.values().iterator()));
91      }
