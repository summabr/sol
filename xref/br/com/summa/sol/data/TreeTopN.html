27  /**
28   * A generic collection limited to size <code>N</code>. If more than <code>N</code>
29   * elements are added, it will attempt to keep the greatest <code>N</code> elements
30   * only, discarding the others. However if there are more elements "tied" at the
31   * <code>N</code>th position, they will be also kept in the collection, up to a strict
32   * limit of <code>max</code> elements.<br>
33   * <br>
34   * Duplicate elements are allowed. For 2 arbitrary elements <code>x</code> and
35   * <code>y</code> such that <code>x.compareTo(y) == 0</code>, the oldest element
36   * in this collection will be considered greater for the purpose of sorting, and
37   * perhaps also for the purpose of discarding (only when necessary to prevent more
38   * than <code>max</code> elements in this collection).<br>
39   * <br>
40   * This collection is optimized for large values of <code>N</code>, keeping its
41   * elements stored in a <code>TreeMap</code> of <code>LinkedList</code>s. For
42   * small values of <code>N</code>, use {@link ListTopN} instead, unless the
43   * distinction between <code>N</code> and <code>max</code> provided by
44   * {@link TreeTopN} is really needed.
45   *
46   * @param <E> The type of elements stored in this collection
47   *
48   * @author Einar Saukas
49   */
50  public class TreeTopN<E extends Comparable<E>> implements TopN<E> {
51      private static final long serialVersionUID = 1L;
52  
53      private int n;
54      private int max;
55      private TreeMap<E, LinkedList<E>> data = new TreeMap<E, LinkedList<E>>();
56      private int size = 0;

58      /**
59       * Constructs an empty collection limited strictly to (greatest) <code>N</code>
60       * elements. This works exactly like {@link TreeTopN#TreeTopN(int, int)} when
61       * <code>max = N</code>.
62       *
63       * @param n Strict limit <code>N</code> on the number of elements stored in this collection
64       */
65      public TreeTopN(int n) {
66          this(n, n);
67      }

69      /**
70       * Constructs an empty collection limited to (greatest) <code>N</code> elements. If
71       * there are multiple elements "tied" at the <code>N</code>th position, they will be
72       * also kept in the collection, up to a strict limit of <code>max</code> elements.
73       * Technically, only elements strictly lower than the first <code>N</code> elements
74       * (when sorted in descending order) will be discarded, up to a maximum limit of
75       * <code>max</code> elements. Therefore this collection may get considerably larger
76       * than size <code>N</code>, depending on the specified limit <code>max</code> and
77       * the amount of "repetition" (regarding element comparison). Moreover, oldest
78       * elements in this collection will be considered greater than other "repeated"
79       * elements for the purpose of ordering, and perhaps also for the purpose of
80       * discarding (only when necessary to prevent more than <code>max</code> elements
81       * in this collection).<br>
82       * <br>
83       * When <code>max = N</code> this collection will enforce a strict limit of
84       * <code>N</code> (greatest) elements, discarding the others. Therefore it will
85       * work exactly like {@link ListTopN} except it will be more efficient for larger
86       * values of <code>N</code>.
87       *
88       * @param n Intended limit <code>N</code> on the number of elements stored in this collection
89       * @param max Strict maximum limit on the number of elements stored in this collection
90       */
91      public TreeTopN(int n, int max) {
92          this.n = n;
93          this.max = max;
94      }

96      private void insertSorted(E elem) {
97          List<E> list = data.get(elem);
98          if (list != null) {
99              list.add(elem);
100         } else {
101             data.put(elem, new LinkedList<E>());
102         }
103         size++;
104     }

106     /**
107      * Attempts to add the specified element to the collection. If the collection
108      * already contains (at least) <code>N</code> elements, and the specified new
109      * element is lower than all of them, it won't be added. Moreover, if the
110      * collection already contains <code>max</code> elements, and the specified
111      * new element is not greater than any of them, it won't be added either.
112      * Otherwise the specified element will be added, and other (lower) element(s)
113      * may be automatically removed in this case.
114      *
115      * @param elem Element to be added
116      *
117      * @return <code>true</code> if this collection changed as a result of the call
118      * (thus if the element was successfully added), <code>false</code> otherwise
119      */
120     @Override
121     public boolean add(E elem) {
122         if (size >= n) {
123             Entry<E, LinkedList<E>> lowest = data.firstEntry();
124             final int cmp = lowest.getKey().compareTo(elem);
125             if (cmp < 0) {
126                 if (size - lowest.getValue().size() == n) {
127                     size -= lowest.getValue().size()+1;
128                     data.pollFirstEntry();
129                 } else if (size >= max) {
130                     if (lowest.getValue().size() > 0) {
131                         lowest.getValue().pollLast();
132                     } else {
133                         data.pollFirstEntry();
134                     }
135                     size--;
136                 }
137             } else if (cmp > 0 || size >= max) {
138                 return false;
139             }
140         }
141         insertSorted(elem);
142         return true;
143     }

145     /**
146      * Adds all elements from the specified collection to this collection.
147      *
148      * @param coll Collection containing elements to be added
149      *
150      * @return <code>true</code> if this collection changed as a result of the call
151      * (thus if at least one element was successfully added), <code>false</code> otherwise
152      */
153     @Override
154     public boolean addAll(Collection<? extends E> coll) {
155         boolean changed = false;
156         for (E elem : coll) {
157             changed = add(elem) || changed;
158         }
159         return changed;
160     }

162     @Override
163     public Iterator<E> iterator() {
164         return new Iterator<E>() {
165             Iterator<Entry<E, LinkedList<E>>> iter = data.descendingMap().entrySet().iterator();
166             private Iterator<E> subIter = null;
167 
168             @Override
169             public E next() {
170                 if (subIter != null && subIter.hasNext()) {
171                     return subIter.next();
172                 }
173                 Entry<E, LinkedList<E>> entry = iter.next();
174                 subIter = entry.getValue().iterator();
175                 return entry.getKey();
176             }
177 
178             @Override
179             public boolean hasNext() {
180                 return (subIter != null && subIter.hasNext()) || iter.hasNext();
181             }

220      /**
221       * Returns <code>true</code> if this collection contains the specified element.
222       * More formally, returns <code>true</code> if and only if this list contains
223       * at least one element <code>e</code> such that

231      @Override
232      public boolean contains(Object obj) {
233          Entry<E, LinkedList<E>> entry = data.floorEntry((E)obj);
234          return entry != null &&
235                  (entry.getKey().equals(obj) ||
236                          (entry.getKey().compareTo((E)obj) == 0 && entry.getValue().contains(obj)));
237      }

258      /**
259       * Returns an array containing all top <code>N</code> elements in
260       * this collection, sorted in descending order. See
261       * {@link java.util.LinkedList#toArray()}.
262       *
263       * @return An array containing all top <code>N</code> elements in this
264       * collection, sorted in descending order
265       */
266      @Override
267      public Object[] toArray() {
268          Object[] array = new Object[size];
269          int i = 0;
270          for (Entry<E, LinkedList<E>> entry : data.descendingMap().entrySet()) {
271              array[i++] = entry.getKey();
272              for (E elem : entry.getValue()) {
273                  array[i++] = elem;
274              }
275          }
276          return array;
277      }

279      /**
280       * Returns an array containing all top <code>N</code> elements in
281       * this collection, sorted in descending order; the runtime type of the
282       * returned array is that of the specified array.
283       * See {@link java.util.LinkedList#toArray(Object[])}.
284       *
285       * @param array The array into which all top <code>N</code> elements in
286       * this collection are to be stored, if it is big enough; otherwise, a
287       * new array of the same runtime type will be allocated for this purpose
288       *
289       * @return An array containing all top <code>N</code> elements in this
290       * collection, sorted in descending order
291       */
292      @Override
293      public <T> T[] toArray(T[] array) {
294          if (array.length < size) {
295              array = (T[])Array.newInstance(array.getClass().getComponentType(), size);
296          }
297          int i = 0;
298          for (Entry<E, LinkedList<E>> entry : data.descendingMap().entrySet()) {
299              array[i++] = (T)entry.getKey();
300              for (E elem : entry.getValue()) {
301                  array[i++] = (T)elem;
302              }
303          }
304          if (array.length > size) {
305              array[size] = null;
306          }
307          return array;
308      }
