17  package br.com.summa.sol.util;

19  import java.util.Map;
20  
21  /**
22   * Convenience class for working with {@link java.util.Map}s.
23   *
24   * @author Einar Saukas
25   */
26  public final class Maps {
27  
28      /**
29       * Prevents instantiation
30       */
31      private Maps() {
32      }
33  
34      /**
35       * Removes all entries from the first map, that are currently mapped with the same key and value in the second map.
36       * 
37       * @param <K> The type of keys stored in maps
38       * @param <V> The type of values stored in maps
39       * @param minuend The first map, from where elements will be removed
40       * @param subtrahend The second map, whose elements will be removed from first map
41       */
42      public static <K,V> void removeAll(Map<K,V> minuend, Map<K,V> subtrahend) {
43          for (Map.Entry<K,V> entry : subtrahend.entrySet()) {
44              minuend.remove(entry.getKey(), entry.getValue());
45          }
46      }
47  }
